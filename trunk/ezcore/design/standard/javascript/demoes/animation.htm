eZ Core animation test

Simple animations:

Click on the boxes to start/stop animation!

Opacity animation
ez.$('test_1').toggle( {duration:1000, transition: ez.fx.sinoidal}, 'opacity', toggle);
width: 755px

Width animation
ez.$('test_2').toggle( {duration:1000, transition: ez.fx.sinoidal}, {width:400}, toggle);

Margin animation
ez.$('test_3').toggle( {duration:1000, transition: ez.fx.sinoidal}, {marginLeft:250}, toggle);

Height animation
ez.$('test_4').toggle( {duration:1000, transition: ez.fx.sinoidal}, {height:300}, toggle);

Display toggling
ez.$('test_5').hide( {duration:50}, {display:'none'} );

Position animation with relative position
ez.$('test_6').toggle( {duration:800, transition: ez.fx.sinoidal}, {top: -300, left:-500}, toggle);
position:relative; top: 0px; left: 0px;

Position animation with absolute position
ez.$('test_7').toggle( {duration:800, transition: ez.fx.sinoidal}, {top: 0, left:0}, toggle);
position: absolute; top: 750px; left: 20px; width: 200px; height: 200px;

Scrolling animation
ez.$('test_8').toggle( {duration:1000, transition: ez.fx.sinoidal}, {scrollTop: 35}, toggle);
overflow: auto; height: 45px;

Visibility animation
ez.$('test_9').toggle( 1000, {opacity: 0.3, visibility: 'hidden'}, toggle);
visibility: hidden; width: 700px

Visibility animation
ez.$('test_9').toggle( 1000, {opacity: 0.3, display: 'none'}, toggle);
